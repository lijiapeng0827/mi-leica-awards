  <script>
    // 说明：为保证页面完全自包含，三等奖图片以 Base64 嵌入。
    // 你如果要换图片：把 thirdThumb 的 src 换成你新图片的 dataURL 或普通链接。

    const dlg = document.getElementById('dlg');
    const openThird = document.getElementById('openThird');

    function medalClass(award){
      if(award.includes('一')) return 'gold';
      if(award.includes('二')) return 'silver';
      return 'bronze';
    }

    function openFromCard(card){
      const award = card.dataset.award || '获奖作品';
      const title = card.dataset.title || '';
      const author = card.dataset.author || '';
      const note = card.dataset.note || '';
      const img = card.querySelector('img');

      document.getElementById('dlgAward').textContent = award;
      document.getElementById('dlgTitle').textContent = title;
      document.getElementById('dlgAuthor').textContent = `作者：${author}`;
      document.getElementById('dlgNote').textContent = note;

      const medal = document.getElementById('dlgMedal');
      medal.className = 'medal ' + medalClass(award);
      document.getElementById('dlgImg').src = img?.src || '';

      dlg.showModal();
    }

    document.querySelectorAll('.award').forEach(card => {
      card.addEventListener('click', () => openFromCard(card));
      card.style.cursor = 'pointer';
    });

    openThird.addEventListener('click', () => {
      openFromCard(document.getElementById('thirdCard'));
    });

    document.getElementById('closeBtn').addEventListener('click', () => dlg.close());
    dlg.addEventListener('click', (e) => {
      // 点击遮罩关闭
      const rect = dlg.getBoundingClientRect();
      const inDialog = e.clientX >= rect.left && e.clientX <= rect.right && e.clientY >= rect.top && e.clientY <= rect.bottom;
      if(!inDialog) dlg.close();
    });

    document.getElementById('downloadBtn').addEventListener('click', () => {
      alert('在电脑上：右键大图 → 另存为；在手机上：长按图片 → 保存。');
    });
  </script>
